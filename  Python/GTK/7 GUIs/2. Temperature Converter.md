def on_application_activate(app: Gtk.Application) -> None:
	win = Gtk.ApplicationWindow(title = "Temperature Converter")
	child = TemperatureConverter(0.0, 32.0).root_widget
	win.set_child(child)
	app.add_window(win)
	win.present()

if __name__ == '__main__':
	app = Gtk.Application(application_id = APPLICATION_ID)
	app.connect('activate', on_application_activate)
	app.run(None)
